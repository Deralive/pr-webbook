import numpy as np
import matplotlib.pyplot as plt
from ipywidgets import interactive, IntSlider
import matplotlib as mpl

mpl.rcParams['text.usetex'] = False
plt.rcParams['font.family'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

def monte_carlo_integration(n_samples=100):
    """
    可视化两种蒙特卡洛积分方法：随机投点法和平均值法
    """
    # 定义被积函数和积分区间
    a, b = 0, 1
    f = lambda x: np.exp(x)
    exact_value = np.exp(1) - np.exp(0)  # e^1 - e^0 = e - 1
    
    # 生成随机样本
    np.random.seed(42)  # 设置随机种子，确保结果可重现
    x = np.random.uniform(a, b, n_samples)
    y = np.random.uniform(0, np.exp(1), n_samples)  # y范围：[0, e]
    
    # 1. 随机投点法
    f_x = f(x)
    inside = y <= f_x
    n_inside = np.sum(inside)
    integral_hit = (b - a) * np.exp(1) * (n_inside / n_samples)
    error_hit = abs(integral_hit - exact_value)
    
    # 2. 平均值法
    integral_mean = (b - a) * np.mean(f_x)
    error_mean = abs(integral_mean - exact_value)
    
    # 生成精细网格用于绘制函数曲线
    x_fine = np.linspace(a, b, 100)
    y_fine = f(x_fine)
    
    # 绘制图形
    plt.figure(figsize=(10, 4))
    
    # 左图：随机投点法
    plt.subplot(1, 2, 1)
    plt.plot(x_fine, y_fine, 'b-', linewidth=2, label=r'$f(x) = e^x$')
    plt.fill_between(x_fine, y_fine, 0, alpha=0.2, color='blue', 
                    label=f'积分区域 (面积={exact_value:.4f})')
    
    # 绘制投点
    plt.scatter(x[inside], y[inside], color='green', s=10, alpha=0.6, 
               label=f'区域内点: {n_inside}')
    plt.scatter(x[~inside], y[~inside], color='red', s=10, alpha=0.6, 
               label=f'区域外点: {n_samples - n_inside}')
    
    plt.axhline(y=np.exp(1), color='k', linestyle='--', alpha=0.3)
    plt.axhline(y=0, color='k', linewidth=0.5)
    plt.axvline(x=a, color='k', linewidth=0.5)
    plt.axvline(x=b, color='k', linewidth=0.5)
    
    plt.xlabel('x')
    plt.ylabel('y')
    plt.title(f'随机投点法 (样本数={n_samples})')
    plt.grid(True, alpha=0.2)
    plt.legend(fontsize=9)
    
    # 添加投点法结果
    plt.text(0.05, 0.05, 
            f'近似值: {integral_hit:.6f}\n'
            f'精确值: {exact_value:.6f}\n'
            f'绝对误差: {error_hit:.6f}\n'
            f'相对误差: {error_hit/exact_value*100:.2f}%',
            transform=plt.gca().transAxes,
            bbox=dict(facecolor='white', alpha=0.8))
    
    # 右图：平均值法
    plt.subplot(1, 2, 2)
    plt.plot(x_fine, y_fine, 'b-', linewidth=2, label=r'$f(x) = e^x$')
    plt.fill_between(x_fine, y_fine, 0, alpha=0.2, color='blue')
    
    # 绘制样本点的函数值
    plt.scatter(x, f_x, color='purple', s=15, alpha=0.6, 
               label='样本点函数值')
    
    # 绘制平均值水平线
    mean_val = np.mean(f_x)
    plt.axhline(y=mean_val, color='red', linestyle='--', 
               label=f'样本平均值: {mean_val:.4f}')
    plt.fill_between([a, b], [mean_val, mean_val], 0, alpha=0.3, color='red', 
                    label=f'近似面积: {integral_mean:.4f}')
    
    plt.axhline(y=0, color='k', linewidth=0.5)
    plt.axvline(x=a, color='k', linewidth=0.5)
    plt.axvline(x=b, color='k', linewidth=0.5)
    
    plt.xlabel('x')
    plt.ylabel('f(x)')
    plt.title(f'平均值法 (样本数={n_samples})')
    plt.grid(True, alpha=0.2)
    plt.legend(fontsize=9)
    
    # 添加平均值法结果
    plt.text(0.05, 0.05, 
            f'近似值: {integral_mean:.6f}\n'
            f'精确值: {exact_value:.6f}\n'
            f'绝对误差: {error_mean:.6f}\n'
            f'相对误差: {error_mean/exact_value*100:.2f}%',
            transform=plt.gca().transAxes,
            bbox=dict(facecolor='white', alpha=0.8))
    
    plt.tight_layout()
    plt.show()

import numpy as np
import matplotlib.pyplot as plt
from matplotlib.animation import FuncAnimation
from ipywidgets import Button, IntSlider, HBox, VBox
from IPython.display import display, HTML

# 设置中文字体
plt.rcParams['font.family'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

class MonteCarloStepDemo:
    def __init__(self):
        # 积分参数设置
        self.a, self.b = 0, 1  # 积分区间
        self.f = lambda x: np.exp(x)  # 被积函数 f(x) = e^x
        self.exact_value = np.exp(1) - 1  # 精确积分值 (e-1)
        
        # 交互控件
        self.max_samples = IntSlider(
            min=5, max=300, step=5, value=50,
            description='最大样本数'
        )
        self.delay = IntSlider(
            min=100, max=1000, step=50, value=300,
            description='每步延迟(ms)'
        )
        self.start_btn = Button(description='开始逐步计算')
        self.start_btn.on_click(self.start_calculation)
        
        # 初始化图形
        self.fig, (self.ax1, self.ax2) = plt.subplots(1, 2, figsize=(8, 4))
        self.setup_plots()
        
        # 显示控件
        self.ui = VBox([HBox([self.max_samples, self.delay]), self.start_btn])
        display(self.ui)
        
    def setup_plots(self):
        # 绘制理论函数曲线
        x_fine = np.linspace(self.a, self.b, 300)
        y_fine = self.f(x_fine)
        
        # 左图：随机投点法
        self.ax1.plot(x_fine, y_fine, 'b-', linewidth=2, label='理论函数 f(x)=e^x')
        self.ax1.fill_between(x_fine, y_fine, alpha=0.2, color='blue')
        self.ax1.set_title('随机投点法')
        self.ax1.set_xlabel('x')
        self.ax1.set_ylabel('y')
        self.ax1.set_xlim(self.a, self.b)
        self.ax1.set_ylim(0, np.exp(1))
        self.ax1.grid(alpha=0.3)
        self.ax1.legend()
        
        # 右图：平均值法
        self.ax2.plot(x_fine, y_fine, 'b-', linewidth=2, label='理论函数 f(x)=e^x')
        self.ax2.fill_between(x_fine, y_fine, alpha=0.2, color='blue')
        self.ax2.set_title('平均值法')
        self.ax2.set_xlabel('x')
        self.ax2.set_ylabel('f(x)')
        self.ax2.set_xlim(self.a, self.b)
        self.ax2.set_ylim(0, np.exp(1))
        self.ax2.grid(alpha=0.3)
        self.ax2.legend()
        
        # 初始化图形元素
        self.hit_points = self.ax1.scatter([], [], c='green', s=30, alpha=0.6, label='区域内点')
        self.miss_points = self.ax1.scatter([], [], c='red', s=30, alpha=0.6, label='区域外点')
        self.sample_points = self.ax2.scatter([], [], c='purple', s=30, alpha=0.6, label='样本点')
        self.mean_line, = self.ax2.plot([], [], 'r--', linewidth=2, label='当前平均值')
        
        # 结果文本
        self.text_hit = self.ax1.text(0.05, 0.95, '', transform=self.ax1.transAxes,
                                     bbox=dict(facecolor='white', alpha=0.8),
                                     verticalalignment='top')
        self.text_mean = self.ax2.text(0.05, 0.95, '', transform=self.ax2.transAxes,
                                      bbox=dict(facecolor='white', alpha=0.8),
                                      verticalalignment='top')
        
        # 添加图例
        self.ax1.legend(loc='lower right')
        self.ax2.legend(loc='lower right')
        
        plt.tight_layout()
        
    def init_step(self):
        # 初始化每轮计算
        self.all_x = np.random.uniform(self.a, self.b, self.max_samples.value)
        self.all_y = np.random.uniform(0, np.exp(1), self.max_samples.value)
        
        # 重置存储列表
        self.hit_x, self.hit_y = [], []
        self.miss_x, self.miss_y = [], []
        self.sample_x, self.sample_fx = [], []
        
        # 清空图形元素
        self.hit_points.set_offsets(np.empty((0, 2)))
        self.miss_points.set_offsets(np.empty((0, 2)))
        self.sample_points.set_offsets(np.empty((0, 2)))
        self.mean_line.set_data([], [])
        self.text_hit.set_text('')
        self.text_mean.set_text('')
        
        return (self.hit_points, self.miss_points, self.sample_points, self.mean_line)
    
    def step_calculation(self, i):
        # 第i步计算（仅使用前i+1个样本）
        if i >= self.max_samples.value:
            return (self.hit_points, self.miss_points, self.sample_points, self.mean_line)
        
        # 获取当前样本点
        x = self.all_x[i]
        y_rand = self.all_y[i]
        fx = self.f(x)
        
        # 1. 随机投点法计算（逐步累积）
        if y_rand <= fx:
            self.hit_x.append(x)
            self.hit_y.append(y_rand)
        else:
            self.miss_x.append(x)
            self.miss_y.append(y_rand)
        
        # 更新投点显示
        self.hit_points.set_offsets(np.c_[self.hit_x, self.hit_y])
        self.miss_points.set_offsets(np.c_[self.miss_x, self.miss_y])
        
        # 重新计算投点法积分值（基于当前所有点）
        n_total = i + 1
        n_hit = len(self.hit_x)
        hit_area = (self.b - self.a) * np.exp(1) * (n_hit / n_total)
        hit_error = abs(hit_area - self.exact_value)
        
        # 更新投点法文本
        self.text_hit.set_text(
            f'第{i+1}步\n'
            f'总样本: {n_total}\n'
            f'区域内点: {n_hit}\n'
            f'积分估计: {hit_area:.6f}\n'
            f'误差: {hit_error:.6f}'
        )
        
        # 2. 平均值法计算（逐步累积）
        self.sample_x.append(x)
        self.sample_fx.append(fx)
        
        # 更新样本点显示
        self.sample_points.set_offsets(np.c_[self.sample_x, self.sample_fx])
        
        # 重新计算平均值法积分值（基于当前所有点）
        mean_fx = np.mean(self.sample_fx)
        mean_area = (self.b - self.a) * mean_fx
        mean_error = abs(mean_area - self.exact_value)
        
        # 更新平均值线
        self.mean_line.set_data([self.a, self.b], [mean_fx, mean_fx])
        
        # 更新平均值法文本
        self.text_mean.set_text(
            f'第{i+1}步\n'
            f'总样本: {n_total}\n'
            f'函数平均值: {mean_fx:.6f}\n'
            f'积分估计: {mean_area:.6f}\n'
            f'误差: {mean_error:.6f}'
        )
        
        return (self.hit_points, self.miss_points, self.sample_points, self.mean_line)
    
    def start_calculation(self, b):
        # 创建逐步计算的动画
        self.ani = FuncAnimation(
            self.fig, self.step_calculation,
            frames=self.max_samples.value,
            init_func=self.init_step,
            interval=self.delay.value,
            blit=True,
            repeat=False
        )
        
        # 显示动画
        plt.close()
        display(self.ui)
        display(HTML(self.ani.to_jshtml()))

# 运行演示
demo = MonteCarloStepDemo()
